procedure R_K(b0,a0,a1,g:real;Var y:real);
Var
k_1,k_2,k_3,k_4:real;//коэффициенты Р-К
begin
k_1:=dt*(-a0*y+b0*g)/a1;
k_2:=dt*(-a0*(y+k_1/2)+b0*g)/a1;
k_3:=dt*(-a0*(y+k_2/2)+b0*g)/a1;
k_4:=dt*(-a0*(y+k_3)+b0*g)/a1;
y:=y+1/6*(k_1+2*k_2+2*k_3+k_4);
end;

procedure astatism(n:byte);
begin
t:=0;

y3:=0;

end;
end;
