get_values <- function(mu){
#Выберем моменты движения заявок и отберем те из них, которые двигаются по нужному узлу:
arrivals <- get_mon_arrivals(env, per_resource = TRUE) %>%
subset(resource %in% mu, select=c("name", "resource","start_time","end_time","activity_time"))
arr<-arrivals
#Получим среднее время нахождения заявки в узле:
cat("Cреднее время нахождения заявки в узле: ")
arr.time <- arr$end_time - arr$start_time
print(mean(arr.time))
#Получим среднее время нахождения заявки в очереди:
cat("Cреднее время нахождения заявки в очереди:")
t_server <- mean(arr.time-arr$activity_time)
