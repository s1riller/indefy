public static int inputInt() {
Scanner sc = new Scanner(System.in);
int number;
do {
while (!sc.hasNextInt()) {
System.out.println("That not a number!");
sc.next(); // this is important!
}
number = sc.nextInt();
} while (number <= 0);
return number;
}
