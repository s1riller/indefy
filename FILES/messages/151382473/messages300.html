CREATE TABLE Faculties (
id_faculty int NOT NULL PRIMARY KEY,
name_faculty varchar(70) NOT NULL UNIQUE,
surname_dean varchar(20) NOT NULL,
name_dean varchar(15) NOT NULL,
thirdname_dean varchar(15));

CREATE TABLE Specializations (
id_spec int NOT NULL PRIMARY KEY,
code_spec varchar(10) NOT NULL UNIQUE,
name_spec varchar(50) NOT NULL UNIQUE,
qualification_spec varchar(50) NOT NULL,
id_faculty int NOT NULL,
FOREIGN KEY (id_faculty) REFERENCES Faculties(id_faculty)
ON DELETE NO ACTION
ON UPDATE CASCADE);

CREATE TABLE Groups (
id_group int NOT NULL PRIMARY KEY,
code_group varchar(5) NOT NULL,
year_group int check (year_group between 1990 and 2022) NOT NULL,
num_group int DEFAULT 1 NOT NULL,
id_spec int NOT NULL,
s_y_n varchar(12) computed by (code_group || '-'|| cast(year_group as char(4)) || '-' || cast(num_group as char(1))),
FOREIGN KEY (id_spec) REFERENCES Specializations(id_spec)
ON DELETE NO ACTION
ON UPDATE CASCADE);

CREATE TABLE Students (
id_student int NOT NULL PRIMARY KEY,
num_zachetki_student int NOT NULL UNIQUE,
surname_student varchar(20) NOT NULL,
name_student varchar(15) NOT NULL,
thirdname_student varchar(15),
salary_father_student int DEFAULT 0 check (salary_father_student >= 0),
salary_mother_student int DEFAULT 0 check (salary_mother_student >= 0),
num_family_members_student int check (num_family_members_student > 0) NOT NULL,
average_salary computed by ((salary_father_student + salary_mother_student) / num_family_members_student),
id_group int NOT NULL,
FOREIGN KEY (id_group) REFERENCES Groups(id_group)
ON DELETE NO ACTION
ON UPDATE CASCADE);

CREATE unique INDEX Ind_Groups on Groups
(
code_group, year_group, num_group
);
